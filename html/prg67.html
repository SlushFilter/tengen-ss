********************************************************************************
* F_SecondsTimer                                                               *
*                                                                              *
* Checks the frame counter for #$3C (60 frames) to count out seconds.          *
* Every second $00 is decremented by 1, and when $00 reaches zero, this        *
* funciton returns a 1.                                                        *
*                                                                              *
* #Arguments                                                                   *
* ($00) : Number of seconds to time.                                           *
*                                                                              *
* #Returns                                                                     *
* A : (0) Timer still ticking (1) Timer expired.                               *
********************************************************************************

                cmp     #$3c
                bne     @return_0         ; Return A = 0 if $0B != #$3C
                lda     #$00              ; Otherwise Set $0B = #$00 ..
                sta     FrameCounter      ; .. and decrement #$00 ..
                dec     Var_00
                beq     @return_1         ; .. then return A = 1
@return_0       lda     #$00
                rts                       ; Return 0

@return_1       lda     #$01
                rts                       ; Return 1
